**Goal:** Create `TSimpleQuerySupabase`, a new `iSimpleQuery` driver that connects SimpleORM to Supabase via its PostgREST REST API.

**Architecture:** The driver follows the exact same pattern as `SimpleQueryHorse.pas` — it implements `iSimpleQuery`, stores SQL in `TStringList`, params in `TParams`, results in `TClientDataSet`. It parses the SQL generated by `TSimpleSQL<T>` and converts it to Supabase PostgREST HTTP calls. Key difference from Horse driver: Supabase uses query-param filters (`?field=eq.value`), PATCH for updates, and requires `apikey` + `Authorization` headers.

**Tech Stack:** Delphi Object Pascal, `System.Net.HttpClient` (native HTTP), `TClientDataSet` (in-memory DataSet), DUnit (tests)

function TSimpleQuerySupabase.ExtractTableName(const aSQL: string): string;
var
  LWords: TArray<string>;
  I: Integer;
  LUpper: string;
begin
  Result := '';
  LWords := aSQL.Trim.Split([' ', #13, #10, #9], TStringSplitOptions.ExcludeEmpty);

  if Length(LWords) < 2 then
    Exit;

  LUpper := UpperCase(LWords[0]);

  if SameText(LUpper, 'SELECT') or SameText(LUpper, 'DELETE') then
  begin
    for I := 1 to High(LWords) do
    begin
      if SameText(LWords[I], 'FROM') and (I + 1 <= High(LWords)) then
      begin
        Result := LWords[I + 1];
        Exit;
      end;
    end;
  end
  else if SameText(LUpper, 'INSERT') then
  begin
    for I := 1 to High(LWords) do
    begin
      if SameText(LWords[I], 'INTO') and (I + 1 <= High(LWords)) then
      begin
        Result := LWords[I + 1];
        Exit;
      end;
    end;
  end
  else if SameText(LUpper, 'UPDATE') then
  begin
    Result := LWords[1];
  end;
end;

function TSimpleQuerySupabase.DetectOperation(const aSQL: string): string;
var
  LTrimmed: string;
begin
  LTrimmed := aSQL.TrimLeft;
  if LTrimmed.ToUpper.StartsWith('INSERT') then
    Result := 'INSERT'
  else if LTrimmed.ToUpper.StartsWith('UPDATE') then
    Result := 'UPDATE'
  else if LTrimmed.ToUpper.StartsWith('DELETE') then
    Result := 'DELETE'
  else
    Result := 'SELECT';
end;
```

function TSimpleQuerySupabase.ExtractInsertFields(const aSQL: string): TArray<string>;
var
  LStart, LEnd: Integer;
  LFieldsStr: string;
  LFields: TArray<string>;
  I: Integer;
begin
  Result := nil;
  LStart := Pos('(', aSQL);
  LEnd := Pos(')', aSQL);

  if (LStart = 0) or (LEnd = 0) or (LEnd <= LStart) then
    Exit;

  LFieldsStr := Copy(aSQL, LStart + 1, LEnd - LStart - 1);
  LFields := LFieldsStr.Split([',']);

  SetLength(Result, Length(LFields));
  for I := 0 to High(LFields) do
    Result[I] := LFields[I].Trim;
end;

**Step 4: Implement ExtractSelectFields**

Extracts field list from `SELECT F1, F2, F3 FROM TABLE`. Returns Supabase `select` param format:

```pascal
function TSimpleQuerySupabase.ExtractSelectFields(const aSQL: string): string;
var
  LWords: TArray<string>;
  I, LFromIdx: Integer;
  LFields: TStringList;
  LUpper: string;
begin
  Result := '*';
  LWords := aSQL.Trim.Split([' ', #13, #10, #9], TStringSplitOptions.ExcludeEmpty);

  if Length(LWords) < 2 then
    Exit;

  LUpper := UpperCase(LWords[0]);
  if not SameText(LUpper, 'SELECT') then
    Exit;

  // Find FROM position
  LFromIdx := -1;
  for I := 1 to High(LWords) do
  begin
    if SameText(LWords[I], 'FROM') then
    begin
      LFromIdx := I;
      Break;
    end;
  end;

  if LFromIdx <= 1 then
    Exit;

  // Check for SELECT * or SELECT FIRST N SKIP N *
  // Firebird pagination: SELECT FIRST N SKIP N fields FROM table
  LFields := TStringList.Create;
  try
    LFields.StrictDelimiter := True;
    LFields.Delimiter := ',';
    for I := 1 to LFromIdx - 1 do
    begin
      // Skip Firebird pagination keywords
      if SameText(LWords[I], 'FIRST') or SameText(LWords[I], 'SKIP') then
        Continue;
      // Skip numeric values (pagination values)
      if TryStrToInt(LWords[I], LFromIdx) then
      begin
        LFromIdx := -1; // Reset, we used it as temp
        for I := 1 to High(LWords) do
          if SameText(LWords[I], 'FROM') then
          begin
            LFromIdx := I;
            Break;
          end;
        Continue;
      end;

      LFields.Add(LWords[I].Replace(',', '').Trim);
    end;

    if (LFields.Count = 1) and (LFields[0] = '*') then
      Result := '*'
    else if LFields.Count > 0 then
    begin
      Result := '';
      for I := 0 to LFields.Count - 1 do
      begin
        if I > 0 then
          Result := Result + ',';
        Result := Result + LFields[I].ToLower;
      end;
    end;
  finally
    LFields.Free;
  end;
end;
```

**Step 5: Implement ExtractWhereFilters**

Converts `WHERE FIELD = :FIELD` to Supabase query param `field=eq.value`:

```pascal
function TSimpleQuerySupabase.ExtractWhereFilters(const aSQL: string): string;
var
  LWherePos: Integer;
  LWhereClause: string;
  LParts: TArray<string>;
  LFieldName, LParamName: string;
  I: Integer;
  LParamValue: Variant;
  LFilters: TStringList;
begin
  Result := '';
  LWherePos := Pos('WHERE', UpperCase(aSQL));
  if LWherePos = 0 then
    Exit;

  LWhereClause := Copy(aSQL, LWherePos + 5, Length(aSQL));

  // Remove ORDER BY, GROUP BY, LIMIT, OFFSET etc from WHERE clause
  LParts := TArray<string>.Create('ORDER', 'GROUP', 'LIMIT', 'OFFSET', 'FIRST', 'SKIP');
  for I := 0 to High(LParts) do
  begin
    var LPos := Pos(LParts[I], UpperCase(LWhereClause));
    if LPos > 0 then
      LWhereClause := Copy(LWhereClause, 1, LPos - 1);
  end;

  LFilters := TStringList.Create;
  try
    // Split by AND
    LParts := LWhereClause.Split(['AND'], TStringSplitOptions.ExcludeEmpty);
    for I := 0 to High(LParts) do
    begin
      LFieldName := '';
      LParamName := '';

      // Parse "FIELD = :PARAM" pattern
      var LCondition := LParts[I].Trim;
      var LEqPos := Pos('=', LCondition);
      if LEqPos > 0 then
      begin
        LFieldName := Copy(LCondition, 1, LEqPos - 1).Trim;
        LParamName := Copy(LCondition, LEqPos + 1, Length(LCondition)).Trim;

        // Remove leading colon from param name
        if LParamName.StartsWith(':') then
          LParamName := LParamName.Substring(1).Trim;

        // Find param value
        LParamValue := Null;
        var J: Integer;
        for J := 0 to FParams.Count - 1 do
        begin
          if SameText(FParams[J].Name, LParamName) then
          begin
            LParamValue := FParams[J].Value;
            Break;
          end;
        end;

        if not VarIsNull(LParamValue) and not VarIsEmpty(LParamValue) then
          LFilters.Add(LFieldName.ToLower + '=eq.' + VarToStr(LParamValue));
      end;
    end;

    if LFilters.Count > 0 then
    begin
      Result := '';
      for I := 0 to LFilters.Count - 1 do
      begin
        if I > 0 then
          Result := Result + '&';
        Result := Result + LFilters[I];
      end;
    end;
  finally
    LFilters.Free;
  end;
end;
```

function TSimpleQuerySupabase.ExtractPagination(const aSQL: string; out aLimit, aOffset: Integer): Boolean;
var
  LWords: TArray<string>;
  I: Integer;
begin
  Result := False;
  aLimit := 0;
  aOffset := 0;

  LWords := aSQL.Trim.Split([' ', #13, #10, #9], TStringSplitOptions.ExcludeEmpty);

  for I := 0 to High(LWords) do
  begin
    // Firebird: SELECT FIRST N SKIP N ...
    if SameText(LWords[I], 'FIRST') and (I + 1 <= High(LWords)) then
    begin
      TryStrToInt(LWords[I + 1], aLimit);
      Result := True;
    end;
    if SameText(LWords[I], 'SKIP') and (I + 1 <= High(LWords)) then
    begin
      TryStrToInt(LWords[I + 1], aOffset);
      Result := True;
    end;
    // MySQL/SQLite: ... LIMIT N OFFSET N
    if SameText(LWords[I], 'LIMIT') and (I + 1 <= High(LWords)) then
    begin
      TryStrToInt(LWords[I + 1], aLimit);
      Result := True;
    end;
    if SameText(LWords[I], 'OFFSET') and (I + 1 <= High(LWords)) then
    begin
      TryStrToInt(LWords[I + 1], aOffset);
      Result := True;
    end;
  end;
end;

function TSimpleQuerySupabase.DoHTTPRequest(const aMethod, aURL: string; const aBody: string): string;
var
  LClient: THTTPClient;
  LResponse: IHTTPResponse;
  LContent: TStringStream;
  LAuthToken: string;
begin
  Result := '';
  LClient := THTTPClient.Create;
  try
    LClient.ContentType := 'application/json';
    LClient.CustomHeaders['apikey'] := FAPIKey;

    if FToken <> '' then
      LAuthToken := FToken
    else
      LAuthToken := FAPIKey;
    LClient.CustomHeaders['Authorization'] := 'Bearer ' + LAuthToken;
    LClient.CustomHeaders['Prefer'] := 'return=representation';

    if SameText(aMethod, 'GET') then
    begin
      LResponse := LClient.Get(aURL);
    end
    else if SameText(aMethod, 'POST') then
    begin
      LContent := TStringStream.Create(aBody, TEncoding.UTF8);
      try
        LResponse := LClient.Post(aURL, LContent);
      finally
        LContent.Free;
      end;
    end
    else if SameText(aMethod, 'PATCH') then
    begin
      LContent := TStringStream.Create(aBody, TEncoding.UTF8);
      try
        LResponse := LClient.Patch(aURL, LContent);
      finally
        LContent.Free;
      end;
    end
    else if SameText(aMethod, 'DELETE') then
    begin
      LResponse := LClient.Delete(aURL);
    end;

    if Assigned(LResponse) then
    begin
      if (LResponse.StatusCode >= 400) then
        raise Exception.CreateFmt('Supabase HTTP %d: %s', [LResponse.StatusCode, LResponse.ContentAsString(TEncoding.UTF8)]);
      Result := LResponse.ContentAsString(TEncoding.UTF8);
    end;
  finally
    LClient.Free;
  end;
end;

function TSimpleQuerySupabase.ParamsToJSON(const aFields: TArray<string>): string;
var
  LObj: TJSONObject;
  I, J: Integer;
  LParam: TParam;
  LInclude: Boolean;
begin
  LObj := TJSONObject.Create;
  try
    for I := 0 to FParams.Count - 1 do
    begin
      LParam := FParams[I];

      // If fields list specified, only include matching fields
      if Length(aFields) > 0 then
      begin
        LInclude := False;
        for J := 0 to High(aFields) do
        begin
          if SameText(LParam.Name, aFields[J]) then
          begin
            LInclude := True;
            Break;
          end;
        end;
        if not LInclude then
          Continue;
      end;

      if VarIsNull(LParam.Value) or VarIsEmpty(LParam.Value) then
        LObj.AddPair(LParam.Name.ToLower, TJSONNull.Create)
      else
      begin
        case LParam.DataType of
          ftInteger, ftSmallint, ftWord, ftLargeint, ftAutoInc, ftShortint:
            LObj.AddPair(LParam.Name.ToLower, TJSONNumber.Create(LParam.AsInteger));
          ftFloat, ftCurrency, ftBCD, ftFMTBcd, ftExtended, ftSingle:
            LObj.AddPair(LParam.Name.ToLower, TJSONNumber.Create(LParam.AsFloat));
          ftBoolean:
            LObj.AddPair(LParam.Name.ToLower, TJSONBool.Create(LParam.AsBoolean));
        else
          LObj.AddPair(LParam.Name.ToLower, LParam.AsString);
        end;
      end;
    end;
    Result := LObj.ToString;
  finally
    LObj.Free;
  end;
end;

procedure TSimpleQuerySupabase.CreateFieldsFromJSONObject(aObj: TJSONObject);
var
  I: Integer;
  LPair: TJSONPair;
begin
  for I := 0 to aObj.Count - 1 do
  begin
    LPair := aObj.Pairs[I];
    if LPair.JsonValue is TJSONNumber then
      FDataSet.FieldDefs.Add(LPair.JsonString.Value, ftFloat, 0, False)
    else if (LPair.JsonValue is TJSONBool) then
      FDataSet.FieldDefs.Add(LPair.JsonString.Value, ftBoolean, 0, False)
    else
      FDataSet.FieldDefs.Add(LPair.JsonString.Value, ftString, 255, False);
  end;
end;

procedure TSimpleQuerySupabase.JSONArrayToDataSet(aArray: TJSONArray);
var
  I, J: Integer;
  LObj: TJSONObject;
  LPair: TJSONPair;
begin
  if aArray.Count = 0 then
    Exit;

  if not (aArray.Items[0] is TJSONObject) then
    Exit;

  CreateFieldsFromJSONObject(TJSONObject(aArray.Items[0]));
  FDataSet.CreateDataSet;

  for I := 0 to aArray.Count - 1 do
  begin
    if not (aArray.Items[I] is TJSONObject) then
      Continue;

    LObj := TJSONObject(aArray.Items[I]);
    FDataSet.Append;
    for J := 0 to LObj.Count - 1 do
    begin
      LPair := LObj.Pairs[J];
      if FDataSet.FindField(LPair.JsonString.Value) <> nil then
      begin
        if LPair.JsonValue is TJSONNull then
          FDataSet.FieldByName(LPair.JsonString.Value).Clear
        else if LPair.JsonValue is TJSONNumber then
          FDataSet.FieldByName(LPair.JsonString.Value).AsFloat := TJSONNumber(LPair.JsonValue).AsDouble
        else if LPair.JsonValue is TJSONBool then
          FDataSet.FieldByName(LPair.JsonString.Value).AsBoolean := TJSONBool(LPair.JsonValue).AsBoolean
        else
          FDataSet.FieldByName(LPair.JsonString.Value).AsString := LPair.JsonValue.Value;
      end;
    end;
    FDataSet.Post;
  end;

  FDataSet.First;
end;

Builds GET URL with `select`, `where` filters, and pagination:

function TSimpleQuerySupabase.Open: iSimpleQuery;
var
  LTableName, LURL, LResponse, LSelect, LFilters: string;
  LLimit, LOffset: Integer;
  LHasParams: Boolean;
begin
  Result := Self;
  LTableName := ExtractTableName(FSQL.Text);
  LURL := BuildSupabaseURL(LTableName);

  LHasParams := False;

  // Add select fields
  LSelect := ExtractSelectFields(FSQL.Text);
  if (LSelect <> '') and (LSelect <> '*') then
  begin
    LURL := LURL + '?select=' + LSelect;
    LHasParams := True;
  end;

  // Add WHERE filters
  LFilters := ExtractWhereFilters(FSQL.Text);
  if LFilters <> '' then
  begin
    if LHasParams then
      LURL := LURL + '&' + LFilters
    else
      LURL := LURL + '?' + LFilters;
    LHasParams := True;
  end;

  // Add pagination
  if ExtractPagination(FSQL.Text, LLimit, LOffset) then
  begin
    if LLimit > 0 then
    begin
      if LHasParams then
        LURL := LURL + '&limit=' + IntToStr(LLimit)
      else
      begin
        LURL := LURL + '?limit=' + IntToStr(LLimit);
        LHasParams := True;
      end;
    end;
    if LOffset > 0 then
    begin
      if LHasParams then
        LURL := LURL + '&offset=' + IntToStr(LOffset)
      else
      begin
        LURL := LURL + '?offset=' + IntToStr(LOffset);
        LHasParams := True;
      end;
    end;
  end;

  LResponse := DoHTTPRequest('GET', LURL);
  JSONToDataSet(LResponse);
end;

function TSimpleQuerySupabase.ExtractUpdateFields(const aSQL: string): TArray<string>;
var
  LSetPos, LWherePos: Integer;
  LSetClause: string;
  LParts: TArray<string>;
  LList: TList<string>;
  I: Integer;
  LField: string;
  LEqPos: Integer;
begin
  Result := nil;
  LSetPos := Pos('SET', UpperCase(aSQL));
  LWherePos := Pos('WHERE', UpperCase(aSQL));

  if LSetPos = 0 then
    Exit;

  if LWherePos > 0 then
    LSetClause := Copy(aSQL, LSetPos + 3, LWherePos - LSetPos - 3)
  else
    LSetClause := Copy(aSQL, LSetPos + 3, Length(aSQL));

  LParts := LSetClause.Split([',']);
  LList := TList<string>.Create;
  try
    for I := 0 to High(LParts) do
    begin
      LEqPos := Pos('=', LParts[I]);
      if LEqPos > 0 then
      begin
        LField := Copy(LParts[I], 1, LEqPos - 1).Trim;
        LList.Add(LField);
      end;
    end;
    Result := LList.ToArray;
  finally
    LList.Free;
  end;
end;

          [LProduto.ID, LProduto.NOME, LProduto.PRECO]))
      else
        Writeln('  Produto nao encontrado');
    finally
      LProduto.Free;
    end;

    // ---- PAGINATION ----
    Writeln('');
    Writeln('=== PAGINACAO (Skip 0, Take 5) ===');
    LProdutos := LDAO.SQL.Skip(0).Take(5).&End.Find;
    try
      Writeln('Produtos (pagina 1): ', LProdutos.Count);
      for LProduto in LProdutos do
        Writeln(SysUtils.Format('  ID: %d | Nome: %s', [LProduto.ID, LProduto.NOME]));
    finally
      LProdutos.Free;
    end;

    // ---- DELETE ----
    Writeln('');
    Writeln('=== DELETE ===');
    LProduto := TProduto.Create;
    try
      LProduto.ID := 1; // Ajuste para um ID existente
      LDAO.Delete(LProduto);
      Writeln('Produto deletado');
    finally
      LProduto.Free;
    end;

  except
    on E: Exception do
      Writeln('ERRO: ', E.Message);
  end;

  Writeln('');
  Writeln('Pressione ENTER para sair...');
  Readln;
end.

# SimpleORM + Supabase

Exemplo de uso do SimpleORM com Supabase como backend via PostgREST API.

## Pre-requisitos

1. Criar conta no [Supabase](https://supabase.com)
2. Criar um novo projeto
3. Criar a tabela `produto` no SQL Editor:

```sql
CREATE TABLE produto (
  id SERIAL PRIMARY KEY,
  nome VARCHAR(100) NOT NULL,
  preco NUMERIC(10,2) DEFAULT 0,
  quantidade INTEGER DEFAULT 0
);
```

4. Copiar a URL do projeto e a API Key (anon ou service_role):
   - Settings > API > Project URL
   - Settings > API > Project API keys

## Como executar

1. Abrir `SimpleORMSupabase.dpr` na IDE Delphi
2. Substituir `SEU-PROJETO` e `SUA-API-KEY` pelos valores do seu projeto
3. Compilar e executar (F9)

## Funcionalidades demonstradas

- **INSERT** - Inserir produto via POST /rest/v1/produto
- **FIND ALL** - Listar todos os produtos via GET /rest/v1/produto
- **UPDATE** - Atualizar produto via PATCH /rest/v1/produto?id=eq.N
- **FIND BY ID** - Buscar produto por ID via GET /rest/v1/produto?id=eq.N
- **PAGINATION** - Paginacao via ?limit=N&offset=N
- **DELETE** - Deletar produto via DELETE /rest/v1/produto?id=eq.N

## Autenticacao

Para usar Row Level Security (RLS), passe o JWT do usuario:

```pascal
LQuery := TSimpleQuerySupabase.New(
  'https://SEU-PROJETO.supabase.co',
  'SUA-API-KEY',
  'JWT-DO-USUARIO'
);
```

## Nota

O driver Supabase usa `System.Net.HttpClient` (nativo do Delphi), sem dependencias externas.
```
